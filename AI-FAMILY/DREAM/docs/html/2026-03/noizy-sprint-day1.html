Create: src/vad.ts
// ─── NOIZYVOX VAD ENGINE ─────────────────────────────────────────
// Voice Activity Detection + Utterance Capture
// Signal: Neumann U87 → UAD Apollo → Core Audio → ffmpeg
// ─────────────────────────────────────────────────────────────────

import { spawn, ChildProcess } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import { v4 as uuidv4 } from 'uuid';

// ── CONFIG ────────────────────────────────────────────────────────
const CONFIG = {
  // !! CHANGE THIS to your Apollo device index from step 1.5 !!
  DEVICE_INDEX: 1,

  SAMPLE_RATE: 48000,
  CHANNELS: 1,
  BIT_DEPTH: 's32',

  // VAD thresholds
  RMS_THRESHOLD_DB: -40,       // speech onset threshold
  SILENCE_TIMEOUT_MS: 800,     // ms of silence before ending utterance
  PRE_BUFFER_MS: 300,          // ms to capture before speech onset
  MIN_UTTERANCE_MS: 500,       // discard utterances shorter than this

  SESSION_DIR: 'sessions',
  LOG_INTERVAL_MS: 200,
};

// ── TYPES ─────────────────────────────────────────────────────────
type VADState = 'silence' | 'speech' | 'speech_ending';

interface Utterance {
  id: string;
  sessionId: string;
  startMs: number;
  endMs: number;
  durationMs: number;
  filePath: string;
  peakRMS: number;
}

// ── VAD ENGINE ────────────────────────────────────────────────────
export class VADEngine {
  private state: VADState = 'silence';
  private sessionId: string;
  private utteranceBuffer: Buffer[] = [];
  private preBuffer: Buffer[] = [];
  private speechStartMs = 0;
  private lastSpeechMs = 0;
  private utteranceCount = 0;
  private ffmpegProc: ChildProcess | null = null;
  private utterances: Utterance[] = [];
  private isRunning = false;
  private onUtterance?: (u: Utterance) => void;

  constructor(sessionId?: string) {
    this.sessionId = sessionId || `SES-RSP001-${Date.now()}`;
    const dir = path.join(CONFIG.SESSION_DIR, this.sessionId);
    fs.mkdirSync(dir, { recursive: true });
    console.log(`[VAD] Session: ${this.sessionId}`);
  }

  private dbToLinear(db: number): number {
    return Math.pow(10, db / 20);
  }

  private computeRMS(buf: Buffer): number {
    let sum = 0;
    const samples = buf.length / 4; // s32 = 4 bytes per sample
    for (let i = 0; i < buf.length - 3; i += 4) {
      const sample = buf.readInt32LE(i) / 2147483648; // normalize to -1..1
      sum += sample * sample;
    }
    return Math.sqrt(sum / Math.max(samples, 1));
  }

  private rmsToDb(rms: number): number {
    return rms > 0 ? 20 * Math.log10(rms) : -100;
  }

  private isSpeech(rms: number): boolean {
    return rms >= this.dbToLinear(CONFIG.RMS_THRESHOLD_DB);
  }

  private processChunk(chunk: Buffer, nowMs: number): void {
    const rms = this.computeRMS(chunk);
    const speaking = this.isSpeech(rms);

    // Maintain pre-buffer (rolling window before speech)
    this.preBuffer.push(chunk);
    const preMs = CONFIG.PRE_BUFFER_MS;
    const bytesPerMs = (CONFIG.SAMPLE_RATE * 4) / 1000;
    while (this.preBuffer.reduce((s, b) => s + b.length, 0) > preMs * bytesPerMs) {
      this.preBuffer.shift();
    }

    switch (this.state) {
      case 'silence':
        if (speaking) {
          this.state = 'speech';
          this.speechStartMs = nowMs;
          this.lastSpeechMs = nowMs;
          this.utteranceBuffer = [...this.preBuffer];
          console.log(`[VAD] ▶ Speech onset — RMS ${this.rmsToDb(rms).toFixed(1)} dBFS`);
        }
        break;

      case 'speech':
        this.utteranceBuffer.push(chunk);
        if (speaking) {
          this.lastSpeechMs = nowMs;
        } else if (nowMs - this.lastSpeechMs > CONFIG.SILENCE_TIMEOUT_MS / 2) {
          this.state = 'speech_ending';
        }
        break;

      case 'speech_ending':
        this.utteranceBuffer.push(chunk);
        if (speaking) {
          this.state = 'speech';
          this.lastSpeechMs = nowMs;
        } else if (nowMs - this.lastSpeechMs > CONFIG.SILENCE_TIMEOUT_MS) {
          this.finalizeUtterance(nowMs);
          this.state = 'silence';
        }
        break;
    }
  }

  private finalizeUtterance(endMs: number): void {
    const durationMs = endMs - this.speechStartMs;
    if (durationMs < CONFIG.MIN_UTTERANCE_MS) {
      console.log(`[VAD] ⚡ Utterance too short (${durationMs}ms) — discarded`);
      this.utteranceBuffer = [];
      return;
    }

    const uttNum = String(++this.utteranceCount).padStart(4, '0');
    const uttId = `UTT-RSP001-${uttNum}`;
    const fileName = `${uttId}_${Date.now()}.wav`;
    const filePath = path.join(CONFIG.SESSION_DIR, this.sessionId, fileName);

    // Combine buffers and write WAV
    const raw = Buffer.concat(this.utteranceBuffer);
    this.writeWAV(filePath, raw);

    const utterance: Utterance = {
      id: uttId,
      sessionId: this.sessionId,
      startMs: this.speechStartMs,
      endMs,
      durationMs,
      filePath,
      peakRMS: this.computeRMS(raw),
    };

    this.utterances.push(utterance);
    this.utteranceBuffer = [];

    console.log(`[VAD] ✓ ${uttId} — ${(durationMs/1000).toFixed(2)}s → ${filePath}`);
    this.onUtterance?.(utterance);
  }

  private writeWAV(filePath: string, pcmData: Buffer): void {
    const numChannels = CONFIG.CHANNELS;
    const sampleRate = CONFIG.SAMPLE_RATE;
    const bitsPerSample = 32;
    const byteRate = sampleRate * numChannels * bitsPerSample / 8;
    const blockAlign = numChannels * bitsPerSample / 8;
    const dataSize = pcmData.length;
    const header = Buffer.alloc(44);

    header.write('RIFF', 0);
    header.writeUInt32LE(36 + dataSize, 4);
    header.write('WAVE', 8);
    header.write('fmt ', 12);
    header.writeUInt32LE(16, 16);
    header.writeUInt16LE(1, 20);  // PCM
    header.writeUInt16LE(numChannels, 22);
    header.writeUInt32LE(sampleRate, 24);
    header.writeUInt32LE(byteRate, 28);
    header.writeUInt16LE(blockAlign, 32);
    header.writeUInt16LE(bitsPerSample, 34);
    header.write('data', 36);
    header.writeUInt32LE(dataSize, 40);

    fs.writeFileSync(filePath, Buffer.concat([header, pcmData]));
  }

  public onNewUtterance(cb: (u: Utterance) => void): void {
    this.onUtterance = cb;
  }

  public async start(): Promise<void> {
    this.isRunning = true;
    const chunkMs = 50;
    const chunkBytes = Math.floor(CONFIG.SAMPLE_RATE * 4 * chunkMs / 1000);

    console.log(`[VAD] Starting capture on device [${CONFIG.DEVICE_INDEX}]`);
    console.log(`[VAD] ${CONFIG.SAMPLE_RATE}Hz / ${CONFIG.BIT_DEPTH} / threshold ${CONFIG.RMS_THRESHOLD_DB}dBFS`);
    console.log(`[VAD] Speak into the U87. Ctrl+C to stop.\n`);

    this.ffmpegProc = spawn('ffmpeg', [
      '-f', 'avfoundation',
      '-i', `:[${CONFIG.DEVICE_INDEX}]`,
      '-ar', String(CONFIG.SAMPLE_RATE),
      '-ac', String(CONFIG.CHANNELS),
      '-sample_fmt', CONFIG.BIT_DEPTH,
      '-f', 's32le',
      'pipe:1',
      '-loglevel', 'error'
    ]);

    let accumulator = Buffer.alloc(0);
    let nowMs = Date.now();

    this.ffmpegProc.stdout?.on('data', (data: Buffer) => {
      accumulator = Buffer.concat([accumulator, data]);
      nowMs = Date.now();
      while (accumulator.length >= chunkBytes) {
        const chunk = accumulator.subarray(0, chunkBytes);
        accumulator = accumulator.subarray(chunkBytes);
        this.processChunk(chunk, nowMs);
      }
    });

    this.ffmpegProc.stderr?.on('data', (d: Buffer) => {
      const msg = d.toString().trim();
      if (msg) console.error(`[ffmpeg] ${msg}`);
    });

    return new Promise((resolve) => {
      this.ffmpegProc?.on('exit', () => {
        if (this.state !== 'silence') this.finalizeUtterance(Date.now());
        console.log(`\n[VAD] Session complete — ${this.utteranceCount} utterances captured`);
        resolve();
      });
    });
  }

  public stop(): void {
    this.isRunning = false;
    this.ffmpegProc?.kill('SIGTERM');
  }

  public getUtterances(): Utterance[] {
    return [...this.utterances];
  }

  public getSessionId(): string {
    return this.sessionId;
  }
}

// ─── NOIZYVOX 7-STAGE PIPELINE ───────────────────────────────────

import * as crypto from 'crypto';

// ── NEVER CLAUSES (NC_01–NC_10) ─────────────────────────────────
const NEVER_CLAUSES = [
  'NC_01: No use without explicit scoped consent',
  'NC_02: No retroactive consent',
  'NC_03: No use in harmful/demeaning content',
  'NC_04: No political advertising without explicit opt-in',
  'NC_05: No synthetic cloning outside consent schema',
  'NC_06: No royalty deferral without consent',
  'NC_07: No training use outside training consent domain',
  'NC_08: No platform sale that changes actor terms',
  'NC_09: No acquisition override of Voice Estate Trust',
  'NC_10: No Estate dissolution without unanimous consent',
];

// ── CONSENT SCHEMA ───────────────────────────────────────────────
interface ConsentKey {
  raw: string;
  actorId: string;
  domain: string;
  jurisdiction: string;
  date: string;
  isValid: boolean;
}

interface PipelineContext {
  utteranceId: string;
  sessionId: string;
  actorId: string;
  filePath: string;
  consentKey?: ConsentKey;
  watermarkHash?: string;
  ncResults?: { id: string; pass: boolean }[];
  ledgerEntry?: ProvenanceRecord;
  timestamp: string;
}

interface ProvenanceRecord {
  utteranceId: string;
  actorId: string;
  sessionId: string;
  consentKey: string;
  watermarkHash: string;
  ncCheck: 'PASS_10/10' | 'FAIL';
  filePath: string;
  timestamp: string;
  pipelineVersion: string;
}

// ── WATERMARK ENGINE ─────────────────────────────────────────────
export function generateWatermark(
  actorId: string,
  utteranceId: string,
  sessionId: string,
  consentKey: string
): string {
  const payload = `${actorId}:${utteranceId}:${sessionId}:${consentKey}:${Date.now()}`;
  const hash = crypto.createHash('sha256').update(payload).digest('hex').toUpperCase();
  return `WM_${hash.slice(0, 8)}`;
}

// ── CONSENT KEY GENERATOR ────────────────────────────────────────
export function generateConsentKey(
  actorId: string,
  domain: string = 'DEMO',
  jurisdiction: string = 'US-CA'
): ConsentKey {
  const hex = crypto.randomBytes(8).toString('hex').toUpperCase();
  const date = new Date().toISOString().slice(0, 10).replace(/-/g, '');
  const raw = `NOIZY-${actorId}-CONSENT-${hex}-AVA-${domain}-${jurisdiction}-${date}`;
  return { raw, actorId, domain, jurisdiction, date, isValid: true };
}

// ── NC CHECKER ───────────────────────────────────────────────────
function checkNeverClauses(ctx: PipelineContext): { id: string; pass: boolean }[] {
  return NEVER_CLAUSES.map((nc, i) => ({
    id: `NC_${String(i + 1).padStart(2, '0')}`,
    pass: ctx.consentKey?.isValid === true, // real logic per clause goes here
  }));
}

// ── 7-STAGE PIPELINE ─────────────────────────────────────────────
export async function runPipeline(
  utteranceId: string,
  sessionId: string,
  filePath: string,
  actorId: string = 'RSP001'
): Promise<PipelineContext> {

  const ctx: PipelineContext = {
    utteranceId, sessionId, actorId, filePath,
    timestamp: new Date().toISOString(),
  };

  console.log(`\n[PIPELINE] Processing ${utteranceId}`);

  // Stage 1: Consent Verification
  ctx.consentKey = generateConsentKey(actorId);
  console.log(`  [S1] Consent: ${ctx.consentKey.raw.slice(0, 40)}...`);

  // Stage 2: Never Clause Check
  ctx.ncResults = checkNeverClauses(ctx);
  const ncAllPass = ctx.ncResults.every(r => r.pass);
  if (!ncAllPass) throw new Error('[PIPELINE] HALT — Never Clause violation');
  console.log(`  [S2] Never Clauses: PASS ${ctx.ncResults.length}/${ctx.ncResults.length}`);

  // Stage 3: WAV Write (already done by VAD — verify file exists)
  const fs = require('fs');
  if (!fs.existsSync(filePath)) throw new Error(`[PIPELINE] WAV not found: ${filePath}`);
  console.log(`  [S3] WAV: ${filePath.split('/').pop()}`);

  // Stage 4: Watermark
  ctx.watermarkHash = generateWatermark(actorId, utteranceId, sessionId, ctx.consentKey.raw);
  console.log(`  [S4] Watermark: ${ctx.watermarkHash}`);

  // Stage 5: Provenance Record
  ctx.ledgerEntry = {
    utteranceId,
    actorId,
    sessionId,
    consentKey: ctx.consentKey.raw,
    watermarkHash: ctx.watermarkHash,
    ncCheck: 'PASS_10/10',
    filePath,
    timestamp: ctx.timestamp,
    pipelineVersion: '1.0.0',
  };
  console.log(`  [S5] Provenance record created`);

  // Stage 6: API Dispatch (stub — real API in sprint extension)
  console.log(`  [S6] API dispatch: QUEUED (demo mode)`);

  // Stage 7: Ledger Write (handled by ProvenanceLedger class)
  console.log(`  [S7] Ready for ledger write`);

  return ctx;
}

// ─── NOIZYVOX PROVENANCE LEDGER ───────────────────────────────────
// Append-only JSONL — never deleted, never modified
// ─────────────────────────────────────────────────────────────────

import * as fs from 'fs';
import * as path from 'path';

const LEDGER_PATH = path.join('ledger', 'provenance.jsonl');

export class ProvenanceLedger {
  constructor() {
    fs.mkdirSync('ledger', { recursive: true });
    if (!fs.existsSync(LEDGER_PATH)) {
      fs.writeFileSync(LEDGER_PATH, '');
      console.log(`[LEDGER] Created: ${LEDGER_PATH}`);
    }
  }

  public append(record: object): void {
    const line = JSON.stringify(record) + '\n';
    fs.appendFileSync(LEDGER_PATH, line);
    const count = this.count();
    console.log(`[LEDGER] Entry #${count} written`);
  }

  public count(): number {
    try {
      const data = fs.readFileSync(LEDGER_PATH, 'utf-8');
      return data.split('\n').filter(l => l.trim()).length;
    } catch { return 0; }
  }

  public getAll(): object[] {
    try {
      return fs.readFileSync(LEDGER_PATH, 'utf-8')
        .split('\n')
        .filter(l => l.trim())
        .map(l => JSON.parse(l));
    } catch { return []; }
  }

  public printSummary(): void {
    const entries = this.getAll() as any[];
    console.log(`\n╔══════════════════════════════════════╗`);
    console.log(`║  NOIZYVOX PROVENANCE LEDGER          ║`);
    console.log(`╠══════════════════════════════════════╣`);
    console.log(`║  Total utterances: ${String(entries.length).padStart(16)} ║`);
    if (entries.length > 0) {
      console.log(`║  Last: ${entries[entries.length-1].utteranceId.padStart(29)} ║`);
      console.log(`║  WM:   ${entries[entries.length-1].watermarkHash.padStart(29)} ║`);
    }

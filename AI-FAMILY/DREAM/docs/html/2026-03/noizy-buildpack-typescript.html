import { AVA } from '../schemas/AVA';
import { ConsentKey } from '../schemas/ConsentKey';
import { ROYALTY_FLOOR_PCT } from './core';

export interface EnforcementResult {
  passed:   boolean;
  violated: string[];
  message:  string;
}

export function enforceNeverClauses(
  ava:        AVA,
  consentKey: ConsentKey | null,
  royaltyPct: number,
  domain:     string,
  inputText:  string
): EnforcementResult {
  const violated: string[] = [];

  // NC_01 — No voice without consent
  if (!consentKey || consentKey.revokedAt) {
    violated.push('NC_01: No valid consent key');
  }

  // NC_02 — No royalty below floor
  if (royaltyPct < ROYALTY_FLOOR_PCT) {
    violated.push(`NC_02: Royalty ${royaltyPct}% below floor ${ROYALTY_FLOOR_PCT}%`);
  }

  // NC_04 — No impersonation (basic check — extend with ML classifier)
  if (ava.prohibitedDomains.includes(domain)) {
    violated.push(`NC_04/NC_05: Domain '${domain}' is prohibited for this AVA`);
  }

  // NC_07 — No political content
  const politicalTerms = ['vote for', 'elect', 'campaign', 'political party'];
  if (politicalTerms.some(t => inputText.toLowerCase().includes(t))) {
    violated.push('NC_07: Political persuasion content detected');
  }

  // NC_08 — No medical advice
  const medicalTerms = ['diagnose', 'prescribe', 'treatment for', 'take this medication'];
  if (medicalTerms.some(t => inputText.toLowerCase().includes(t))) {
    violated.push('NC_08: Medical advice synthesis detected');
  }

  return {
    passed:   violated.length === 0,
    violated,
    message:  violated.length === 0
      ? 'All Never Clauses passed.'
      : `Blocked by: ${violated.join(' | ')}`,
  };
}

import { ConsentKey } from '../../packages/schemas/ConsentKey';
import { generateConsentKey } from './keyGenerator';

export class ConsentService {
  constructor(private db: D1Database) {}

  // Issue a new consent key for an Actor/AVA pair
  async issue(
    actorId:      string,
    avaId:        string,
    domain:       string,
    jurisdiction: string
  ): Promise<ConsentKey> {
    const keyId = generateConsentKey(actorId, domain, jurisdiction);
    const key: ConsentKey = {
      keyId, actorId, avaId, domain, jurisdiction,
      issuedAt:            new Date().toISOString(),
      neverClausesApplied: ['NC_01','NC_02','NC_03','NC_10'],
      hash:                await sha256(keyId),
    };
    await this.db.prepare(
      'INSERT INTO consent_keys VALUES (?,?,?,?,?,?,?,?,?)'
    ).bind(...Object.values(key)).run();
    return key;
  }

  // Validate a consent key — called at Stage 1 of every synthesis
  async validate(keyId: string): Promise<ConsentKey | null> {
    const result = await this.db
      .prepare('SELECT * FROM consent_keys WHERE keyId = ? AND revokedAt IS NULL')
      .bind(keyId)
      .first<ConsentKey>();
    return result ?? null;
  }

  // Revoke — NC_10: marks revokedAt, NEVER deletes the record
  async revoke(keyId: string, actorId: string): Promise<void> {
    await this.db
      .prepare('UPDATE consent_keys SET revokedAt = ? WHERE keyId = ? AND actorId = ?')
      .bind(new Date().toISOString(), keyId, actorId)
      .run();
  }
}

async function sha256(text: string): Promise<string> {
  const buf = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(text));
  return [...new Uint8Array(buf)].map(b => b.toString(16).padStart(2,'0')).join('');
}

// Stage 4: Generate NOIZY PROOF watermark WM_{8hex}
// SHA-256(actorId:utteranceId:sessionId:consentKeyId:timestamp)

export async function generateWatermark(params: {
  actorId:      string;
  utteranceId:  string;
  sessionId:    string;
  consentKeyId: string;
  timestamp:    string;
}): Promise<string> {
  const seed = [
    params.actorId,
    params.utteranceId,
    params.sessionId,
    params.consentKeyId,
    params.timestamp,
  ].join(':');

  const buf = await crypto.subtle.digest(
    'SHA-256',
    new TextEncoder().encode(seed)
  );

  const hex = [...new Uint8Array(buf)]
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');

  return `WM_${hex.slice(0, 8).toUpperCase()}`;
}

// Verify a watermark against known params
export async function verifyWatermark(
  watermarkId: string,
  params: Parameters<typeof generateWatermark>[0]
): Promise<boolean> {
  const expected = await generateWatermark(params);
  return expected === watermarkId;
}

import { ROYALTY_FLOOR_PCT, FOUNDING_RSP_ROYALTY_PCT } from '../../packages/never-clauses/core';
import { Actor } from '../../packages/schemas/Actor';

export interface RoyaltyAllocation {
  actorId:     string;
  actorShare:  number;   // Always >= ROYALTY_FLOOR_PCT
  noizyShare:  number;   // 100 - actorShare
  grossRevenue:number;
  actorPayout: number;
  noizyPayout: number;
  timestamp:   string;
}

export function calculateRoyalty(
  actor:        Actor,
  grossRevenue: number
): RoyaltyAllocation {
  // Enforce NC_02 — floor is constitutional, not configurable
  const actorShare = actor.royaltyTier === 'founding'
    ? FOUNDING_RSP_ROYALTY_PCT   // 85% — locked forever
    : Math.max(actor.royaltyPct, ROYALTY_FLOOR_PCT);  // Never below 75%

  const noizyShare  = 100 - actorShare;
  const actorPayout = (grossRevenue * actorShare) / 100;
  const noizyPayout = (grossRevenue * noizyShare) / 100;

  return {
    actorId: actor.actorId,
    actorShare, noizyShare,
    grossRevenue,
    actorPayout: Math.round(actorPayout * 100) / 100,
    noizyPayout: Math.round(noizyPayout * 100) / 100,
    timestamp: new Date().toISOString(),
  };
}

export async function runSynthesisPipeline(params: {
  actor:       Actor;
  ava:         AVA;

  sessionId:   string;
  domain:      string;
  consentSvc:  ConsentService;
  db:          D1Database;
  r2:          R2Bucket;
}) {
  const { actor, ava, inputText, sessionId, domain, consentSvc } = params;
  const utteranceId = `UTT-${crypto.randomUUID()}`;
  const timestamp   = new Date().toISOString();

  // ── STAGE 1: Validate consent key ──────────────────────────────────────────
  const consentKey = await consentSvc.validate(actor.consentKeyId);
  if (!consentKey) return { ok: false, error: voiceOfRefusal('No valid consent key (NC_01)') };

  // ── STAGE 2: Never Clause enforcement ──────────────────────────────────────
  const check = enforceNeverClauses(ava, consentKey, actor.royaltyPct, domain, inputText);
  if (!check.passed) return { ok: false, error: voiceOfRefusal(check.message, 'educational') };

  // ── STAGE 3: Synthesize WAV ─────────────────────────────────────────────────
  const wavBuffer = await synthesizeVoice(inputText, ava);

  await params.r2.put(wavKey, wavBuffer);

  const watermarkId = await generateWatermark({
    actorId: actor.actorId, utteranceId, sessionId,
    consentKeyId: consentKey.keyId, timestamp
  });

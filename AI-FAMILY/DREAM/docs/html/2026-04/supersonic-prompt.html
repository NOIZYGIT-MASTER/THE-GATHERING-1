git-align.sh · Per-Repo Cutover · ~/NOIZYLAB/git-align.sh

#!/bin/zsh # git-align.sh — NOIZY Enterprise Git Cutover # Doctrine: push enterprise-first, retire GitHub safely
BASE_DIR="$HOME/NOIZYLAB" ORG="NOIZYFISH" ENTERPRISE_HOST="git@git.noizy.ai:${ORG}"
# Safe global defaults git config --global push.default current git config --global fetch.prune true git config --global init.defaultBranch main
# Find every repo under BASE_DIR find "$BASE_DIR" -type d -name ".git" | while read -r gitdir; do   repo_dir="$(dirname "$gitdir")"   repo_name="$(basename "$repo_dir")"   cd "$repo_dir" || continue
  # Rename origin → github (keep as backup)   if git remote get-url origin >/dev/null 2>&1; then     current="$(git remote get-url origin)"     if [[ "$current" != *"git.noizy.ai"* ]]; then       git remote rename origin github 2>/dev/null || true     fi   fi
  # Add or update enterprise remote   if git remote get-url enterprise >/dev/null 2>&1; then     git remote set-url enterprise "${ENTERPRISE_HOST}/${repo_name}.git"   else     git remote add enterprise "${ENTERPRISE_HOST}/${repo_name}.git"   fi
  git config remote.pushDefault enterprise   git remote -v done

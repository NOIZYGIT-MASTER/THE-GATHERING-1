🔥 REAL COMMAND HANDLERS

import * as vscode from "vscode";

export function activate(context: vscode.ExtensionContext) {

  const runCommand = async (intent: string) => {
    const res = await fetch("http://localhost:8787/api/command", {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        intent,
        confirmed: true,
        source: "noizybeast",
        actor: "RSP001"
      })
    });

    const data = await res.json();

    vscode.window.showInformationMessage(
      `${intent} → ${data.summary || data.error}`
    );
  };

  context.subscriptions.push(
    vscode.commands.registerCommand("noizybeast.deployWorker", async () => {
      await runCommand("deploy_system");
    })
  );

  context.subscriptions.push(
    vscode.commands.registerCommand("noizybeast.runTurbo", async () => {
      await runCommand("run_ci");
    })
  );

  context.subscriptions.push(
    vscode.commands.registerCommand("noizybeast.showStatus", async () => {
      const res = await fetch("http://localhost:8787/api/status");
      const data = await res.json();

      vscode.window.showInformationMessage(
        `DNS: ${data.dns} | CF: ${data.cloudflare}`
      );
    })
  );

  context.subscriptions.push(
    vscode.commands.registerCommand("noizybeast.sendToGabriel", async () => {
      await runCommand("deploy_system");
    })
  );
}
